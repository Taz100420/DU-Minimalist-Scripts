--Call switch gears from the input.

groundalt = telemeter.raycast().distance
altitude = core.getAltitude()
targetalt = 20 --export

if groundalt == -1 then
    groundalt = altitude
    end

if groundalt < targetalt then
    gears.activate() else
    gears.deactivate()
    end

if altitude < targetalt then
    gears.deactivate()
    end
